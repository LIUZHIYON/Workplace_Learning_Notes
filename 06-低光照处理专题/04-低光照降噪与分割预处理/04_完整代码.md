## Python 实现 (含所有方法)

```python
"""
low_light_preprocessing.py
低光照图像分割前预处理 - 全部方法实现
支持 OpenCV 4.x+
"""

import cv2
import numpy as np
import time
from dataclasses import dataclass
from typing import Tuple, Optional


# ============================================================
# 基础降噪方法
# ============================================================

def denoise_gaussian(img: np.ndarray, ksize: int = 5, sigma: float = 1.0) -> np.ndarray:
    """高斯滤波降噪 — 最快速但模糊边缘"""
    return cv2.GaussianBlur(img, (ksize, ksize), sigmaX=sigma)

def denoise_median(img: np.ndarray, ksize: int = 3) -> np.ndarray:
    """中值滤波 — 去椒盐噪声，速度好"""
    return cv2.medianBlur(img, ksize)

def denoise_bilateral(img: np.ndarray, d: int = 5, sigma_color: float = 75,
                      sigma_space: float = 75) -> np.ndarray:
    """双边滤波 — 边缘保持降噪"""
    return cv2.bilateralFilter(img, d, sigma_color, sigma_space)

def denoise_nlm(img: np.ndarray, h: float = 10, template_ws: int = 7,
                search_ws: int = 11) -> np.ndarray:
    """非局部均值降噪 — 最高质量"""
    if len(img.shape) == 2:
        return cv2.fastNlMeansDenoising(img, None, h, template_ws, search_ws)
    else:
        return cv2.fastNlMeansDenoisingColored(img, None, h, h, template_ws, search_ws)

def denoise_guided(img: np.ndarray, radius: int = 8, eps: float = 0.01) -> np.ndarray:
    """导向滤波 — O(N) 边缘保持降噪，ARM 首选"""
    eps_scaled = eps * eps * 255 * 255
    return cv2.ximgproc.guidedFilter(img, img, radius, eps_scaled)


# ============================================================
# 对比度增强方法
# ============================================================

def enhance_clahe(img: np.ndarray, clip_limit: float = 1.5,
                  tile_size: int = 8) -> np.ndarray:
    """CLAHE 局部对比度增强"""
    clahe = cv2.createCLAHE(clipLimit=clip_limit, tileGridSize=(tile_size, tile_size))
    if len(img.shape) == 2:
        return clahe.apply(img)
    else:
        lab = cv2.cvtColor(img, cv2.COLOR_BGR2LAB)
        l, a, b = cv2.split(lab)
        l = clahe.apply(l)
        return cv2.cvtColor(cv2.merge([l, a, b]), cv2.COLOR_LAB2BGR)

def enhance_gamma_lut(img: np.ndarray, gamma: float = 0.65) -> np.ndarray:
    """Gamma 校正 (LUT 实现) — 极速"""
    lut = np.array([((i / 255.0) ** gamma) * 255 for i in range(256)], dtype=np.uint8)
    return cv2.LUT(img, lut)

def enhance_gamma(img: np.ndarray, gamma: float = 0.65) -> np.ndarray:
    """Gamma 校正 (逐像素) — 较慢，仅作参考"""
    img_f = img.astype(np.float32) / 255.0
    img_f = np.power(img_f, gamma) * 255
    return img_f.astype(np.uint8)


# ============================================================
# 完整流水线
# ============================================================

def pipeline_fast(img: np.ndarray) -> np.ndarray:
    """
    🥇 极速流水线: 中值 + Gamma + CLAHE
    CPU(720P): ~6-8ms
    """
    denoised = denoise_median(img, 3)
    enhanced = enhance_gamma_lut(denoised, 0.65)
    result = enhance_clahe(enhanced, clip_limit=1.0)
    return result

def pipeline_balanced(img: np.ndarray) -> np.ndarray:
    """
    🥇 平衡流水线: 导向滤波 + Gamma + CLAHE (推荐)
    CPU(720P): ~10-15ms
    """
    denoised = denoise_guided(img, radius=8, eps=0.02)
    enhanced = enhance_gamma_lut(denoised, 0.7)
    result = enhance_clahe(enhanced, clip_limit=1.5)
    return result

def pipeline_quality(img: np.ndarray) -> np.ndarray:
    """
    🥈 质量优先: NLM + 导向 + CLAHE
    CPU(720P): ~50-80ms
    """
    denoised = denoise_nlm(img, h=10, search_ws=11)
    smoothed = denoise_guided(denoised, radius=4, eps=0.02)
    result = enhance_clahe(smoothed, clip_limit=2.0)
    return result


# ============================================================
# 性能测量
# ============================================================

@dataclass
class BenchmarkResult:
    name: str
    cpu_ms: float
    image: np.ndarray

def benchmark(img: np.ndarray, func, name: str, warmup: int = 3,
              measure: int = 10) -> BenchmarkResult:
    """测量函数平均耗时"""
    for _ in range(warmup):
        func(img)

    start = time.perf_counter()
    for _ in range(measure):
        func(img)
    elapsed = (time.perf_counter() - start) / measure * 1000

    return BenchmarkResult(name=name, cpu_ms=elapsed, image=func(img))


def generate_report(img_path: str):
    """生成完整测试报告"""
    img = cv2.imread(img_path)
    if img is None:
        raise FileNotFoundError(f"Cannot read {img_path}")

    print(f"Image: {img_path} ({img.shape[1]}x{img.shape[0]})")
    print(f"{'Method':<35} {'CPU (ms)':>10}")
    print("-" * 47)

    methods = [
        (lambda i: i, "Original (no processing)"),
        (enhance_clahe, "CLAHE only (clip=1.5)"),
        (lambda i: denoise_gaussian(i), "Gaussian Blur only"),
        (lambda i: denoise_median(i), "Median Blur (k=3)"),
        (lambda i: denoise_bilateral(i), "Bilateral (d=5)"),
        (lambda i: denoise_guided(i), "Guided Filter (r=8)"),
        (lambda i: denoise_nlm(i), "NLM (sw=11)"),
        (pipeline_fast, "Pipeline: Median+Gamma+CLAHE"),
        (pipeline_balanced, "Pipeline: Guided+Gamma+CLAHE"),
        (pipeline_quality, "Pipeline: NLM+Guided+CLAHE"),
    ]

    results = []
    for func, name in methods:
        result = benchmark(img, func, name)
        results.append(result)
        print(f"{name:<35} {result.cpu_ms:>8.2f} ms")

    return results
```

## C++ 实现 (RK3576 嵌入式)

```cpp
/**
 * low_light_preprocessing.cpp
 * 低光照分割预处理 - 嵌入式 C++ 版本
 * 编译: g++ -O3 -march=armv8-a -mfpu=neon-vfpv4 -fopenmp
 */

#include <opencv2/opencv.hpp>
#include <opencv2/ximgproc.hpp>

class LowLightPreprocessor {
public:
    LowLightPreprocessor(
        float gamma = 0.65f,
        float clahe_clip = 1.5f,
        int guided_radius = 8,
        float guided_eps = 0.02f
    ) : gamma_(gamma), clahe_clip_(clahe_clip),
        guided_radius_(guided_radius), guided_eps_(guided_eps) {
        init();
    }

    void init() {
        // 预计算 Gamma LUT
        gamma_lut_ = cv::Mat(1, 256, CV_8U);
        uchar* p = gamma_lut_.ptr<uchar>();
        for (int i = 0; i < 256; i++) {
            p[i] = cv::saturate_cast<uchar>(255.0 * pow(i / 255.0, gamma_));
        }

        // 创建 CLAHE 对象（复用）
        clahe_ = cv::createCLAHE(clahe_clip_, cv::Size(8, 8));

        // 预计算导向滤波 eps
        guided_eps_scaled_ = guided_eps_ * guided_eps_ * 255.0 * 255.0;
    }

    // 快速流水线 (推荐)
    void process_fast(const cv::Mat& src, cv::Mat& dst) {
        cv::medianBlur(src, work1_, 3);
        cv::LUT(work1_, gamma_lut_, work2_);
        apply_clahe(work2_, dst);
    }

    // 平衡流水线 (首选)
    void process_balanced(const cv::Mat& src, cv::Mat& dst) {
        cv::ximgproc::guidedFilter(src, src, work1_,
                                   guided_radius_, guided_eps_scaled_);
        cv::LUT(work1_, gamma_lut_, work2_);
        apply_clahe(work2_, dst);
    }

    // 质量流水线
    void process_quality(const cv::Mat& src, cv::Mat& dst) {
        cv::fastNlMeansDenoisingColored(src, work1_,
                                        10.0, 10.0, 7, 11);
        cv::ximgproc::guidedFilter(work1_, work1_, work2_,
                                   4, guided_eps_scaled_);
        apply_clahe(work2_, dst);
    }

    // 仅灰度版本 — 分割常用
    void process_grayscale(const cv::Mat& src, cv::Mat& dst) {
        cv::cvtColor(src, gray_work_, cv::COLOR_BGR2GRAY);
        cv::ximgproc::guidedFilter(gray_work_, gray_work_, work1_,
                                   guided_radius_, guided_eps_scaled_);
        clahe_->apply(work1_, dst);
    }

private:
    void apply_clahe(const cv::Mat& src, cv::Mat& dst) {
        if (src.channels() == 3) {
            cv::cvtColor(src, lab_work_, cv::COLOR_BGR2LAB);
            cv::split(lab_work_, lab_channels_);
            clahe_->apply(lab_channels_[0], lab_channels_[0]);
            cv::merge(lab_channels_, 3, lab_work_);
            cv::cvtColor(lab_work_, dst, cv::COLOR_LAB2BGR);
        } else {
            clahe_->apply(src, dst);
        }
    }

    // 参数
    float gamma_;
    float clahe_clip_;
    int guided_radius_;
    float guided_eps_;
    double guided_eps_scaled_;

    // 预计算对象
    cv::Mat gamma_lut_;
    cv::Ptr<cv::CLAHE> clahe_;

    // 工作缓冲（避免反复分配）
    cv::Mat work1_, work2_, gray_work_, lab_work_;
    std::vector<cv::Mat> lab_channels_;
};

// 使用示例
int main() {
    cv::Mat frame = cv::imread("/dev/video0_frame.jpg");
    cv::Mat processed;

    LowLightPreprocessor preprocessor(0.65f, 1.5f, 8, 0.02f);

    // 循环处理
    while (true) {
        // 从相机获取 frame ...
        // frame = capture_next_frame();

        preprocessor.process_balanced(frame, processed);

        // 送入分割模型 ...
        // segment(processed);
    }
    return 0;
}
```
